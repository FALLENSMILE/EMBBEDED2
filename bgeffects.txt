document.querySelectorAll('.particle').forEach(p => {
    const randomLeft = Math.random() * 100;   // anywhere 0–100%
    const randomDelay = Math.random() * 10;   // 0–10s delay
    const randomDuration = 12 + Math.random() * 10; // 12–22s
    const randomSize = 20 + Math.random() * 40; // 20–60px size

    p.style.setProperty("left", randomLeft + "%", "important");
    p.style.animationDelay = randomDelay + "s";
    p.style.animationDuration = randomDuration + "s";
    p.querySelector("img").style.width = randomSize + "px";
});
